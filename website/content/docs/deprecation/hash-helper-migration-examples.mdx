# {{hash}} Helper Migration Examples

Real-world patterns you'll see in OpenBao UI and how to fix them.

## Tab/Panel State

**Before:**
```handlebars
{{#let (hash activeTab="overview") as |state|}}
  <TabNav @state={{state}} />
  <TabContent @state={{state}} />
{{/let}}
```

**After:**
```handlebars
<TabNav @activeTab={{this.activeTab}} @onSelectTab={{this.selectTab}} />
<TabContent @activeTab={{this.activeTab}} />
```

```javascript
import Component from "@glimmer/component";
import { tracked } from "@ember/object";

export default class TabsComponent extends Component {
  @tracked activeTab = "overview";

  selectTab = (tab) => {
    this.activeTab = tab;
  }
}
```

## Form Input State

**Before:**
```handlebars
{{#let (hash name="" email="" role="viewer") as |form|}}
  <UserForm @data={{form}} />
{{/let}}
```

**After:**
```handlebars
<UserForm @name={{this.name}} @email={{this.email}} @role={{this.role}} @onChange={{this.updateField}} />
```

```javascript
import Component from "@glimmer/component";
import { tracked } from "@ember/object";

export default class UserFormComponent extends Component {
  @tracked name = "";
  @tracked email = "";
  @tracked role = "viewer";

  updateField = (field, value) => {
    this[field] = value;
  }
}
```

## Query/Filter State

**Before:**
```handlebars
{{#let (hash filter="" tab="methods") as |q|}}
  <FilterControl @query={{q}} />
  <Results @query={{q}} />
{{/let}}
```

**After (Parent):**
```handlebars
<FilterControl @filter={{this.filter}} @onFilter={{this.setFilter}} />
<Results @tab={{this.tab}} @filter={{this.filter}} />
```

```javascript
import Component from "@glimmer/component";
import { tracked } from "@ember/object";

export default class SearchComponent extends Component {
  @tracked filter = "";
  @tracked tab = "methods";

  setFilter = (value) => {
    this.filter = value;
  }
}
```

## Array/List Management

**Before:**
```handlebars
{{#let (hash items=(array)) as |container|}}
  <ItemList @items={{container.items}} />
{{/let}}
```

**After:**
```handlebars
<ItemList @items={{this.items}} @onAdd={{this.addItem}} />
```

```javascript
import Component from "@glimmer/component";
import { tracked } from "@ember/object";

export default class ListContainer extends Component {
  @tracked items = [];

  addItem = (item) => {
    this.items = [...this.items, item];
  }
}
```

## Config Objects

**Before:**
```handlebars
{{#let (hash apiUrl="http://..." timeout=5000 retry=3) as |config|}}
  <APIClient @config={{config}} />
{{/let}}
```

**After (option 1 - individual properties):**
```handlebars
<APIClient @apiUrl={{this.config.apiUrl}} @timeout={{this.config.timeout}} @retry={{this.config.retry}} />
```

**After (option 2 - tracking the whole object):**
```handlebars
<APIClient @config={{this.config}} />
```

```javascript
import Component from "@glimmer/component";
import { tracked } from "@ember/object";

export default class APISetupComponent extends Component {
  @tracked config = {
    apiUrl: "http://...",
    timeout: 5000,
    retry: 3
  }
}
```

---

## Quick Checklist

- [ ] Replace hash objects with @tracked properties
- [ ] Pass individual properties to child components
- [ ] Use actions/callbacks for state updates from children
- [ ] Avoid direct mutations of object properties
